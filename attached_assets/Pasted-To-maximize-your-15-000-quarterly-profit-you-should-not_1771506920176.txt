When the TON Pay SDK confirms a payment, the middleware assigns the spins based on the tier_id:

// server/services/membershipService.ts

const SPIN_CONFIG = {
  0: 1,  // Free
  1: 4,  // Bronze
  2: 12, // Silver
  3: 40  // Gold
};

export const activateMembership = async (userId, tierId) => {
  const spins = SPIN_CONFIG[tierId];
  
  await db.users.update(userId, {
    tier_id: tierId,
    spinsRemaining: spins,
    subscription_expiry: addDays(new Date(), 30)
  });
};
