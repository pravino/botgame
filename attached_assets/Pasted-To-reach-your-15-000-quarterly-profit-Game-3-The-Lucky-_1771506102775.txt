export const spinWheel = async (userId, db) => {
  const user = await db.getUser(userId);
  if (user.spinsRemaining <= 0) return { error: "No spins left!" };

  // 1. Check the Jackpot Vault
  const vaultBalance = await db.getVaultBalance('WHEEL_JACKPOT');
  
  // 2. Generate Random Result (0 - 1000)
  const rng = Math.floor(Math.random() * 1000);
  let reward = 0;
  let label = "Better luck next time!";

  if (rng === 777 && vaultBalance >= 100) {

    reward = 0.50;
    label = "Common Win";
  }

  // 3. Atomic Update: Subtract spin, add reward, deduct from vault
  await db.transaction(async (tx) => {
    await tx.update(users).set({ 
      spinsRemaining: user.spinsRemaining - 1,
      virtual_usdt_balance: sql`virtual_usdt_balance + ${reward}`
    }).where(eq(users.id, userId));
    
    if (reward > 0) {
      await tx.update(vaults).set({ 
        balance: sql`balance - ${reward}` 
      }).where(eq(vaults.name, 'WHEEL_JACKPOT'));
    }
  });

  return { reward, label };
